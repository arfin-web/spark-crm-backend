TABLE: users
  - id (UUID, PK)
  - email (VARCHAR, unique)
  - first_name (VARCHAR)
  - last_name (VARCHAR)
  - agency_name (VARCHAR)
  - subscription_tier (VARCHAR: 'starter', 'growth', 'custom')
  - subscription_status (VARCHAR: 'active', 'inactive', 'trial')
  - password (VARCHAR)
  - created_at (TIMESTAMP)
  - updated_at (TIMESTAMP)
  - deleted_at (TIMESTAMP, soft delete)

TABLE: clients
  - id (UUID, PK)
  - user_id (UUID, FK → users)
  - name (VARCHAR)
  - email (VARCHAR)
  - phone (VARCHAR, nullable)
  - company_name (VARCHAR, nullable)
  - industry (VARCHAR, nullable)
  - status (VARCHAR: 'active', 'inactive', 'prospect')
  - health_score (INTEGER: 0-100, auto-calculated from interactions)
  - last_contact_date (TIMESTAMP, nullable)
  - tags (TEXT[], nullable)
  - notes (TEXT, nullable)
  - source (VARCHAR: 'referral', 'website', 'cold_outreach', 'existing', 'other')
  - created_at (TIMESTAMP)
  - updated_at (TIMESTAMP)

TABLE: projects
  - id (UUID, PK)
  - user_id (UUID, FK → users)
  - client_id (UUID, FK → clients)
  - name (VARCHAR)
  - description (TEXT, nullable)
  - status (VARCHAR: 'prospect', 'proposal_sent', 'in_progress', 'completed', 'paused')
  - budget (DECIMAL, nullable)
  - estimated_hours (INTEGER, nullable)
  - start_date (DATE, nullable)
  - end_date (DATE, nullable)
  - created_at (TIMESTAMP)
  - updated_at (TIMESTAMP)

TABLE: activities
  - id (UUID, PK)
  - user_id (UUID, FK → users)
  - client_id (UUID, FK → clients, nullable)
  - project_id (UUID, FK → projects, nullable)
  - type (VARCHAR: 'email', 'call', 'meeting', 'proposal', 'note')
  - title (VARCHAR)
  - description (TEXT)
  - created_at (TIMESTAMP)

TABLE: proposals
  - id (UUID, PK)
  - user_id (UUID, FK → users)
  - client_id (UUID, FK → clients)
  - project_id (UUID, FK → projects, nullable)
  - title (VARCHAR)
  - brief_description (TEXT)
  - scope (TEXT)
  - deliverables (TEXT)
  - timeline (VARCHAR)
  - cost (DECIMAL, nullable)
  - status (VARCHAR: 'draft', 'sent', 'accepted', 'rejected')
  - ai_generated (BOOLEAN, default: true)
  - created_at (TIMESTAMP)
  - updated_at (TIMESTAMP)

TABLE: email_drafts
  - id (UUID, PK)
  - user_id (UUID, FK → users)
  - client_id (UUID, FK → clients, nullable)
  - subject (VARCHAR)
  - body (TEXT)
  - purpose (VARCHAR: 'follow_up', 'proposal', 'check_in', 'custom')
  - status (VARCHAR: 'draft', 'sent')
  - ai_generated (BOOLEAN)
  - created_at (TIMESTAMP)

TABLE: ai_interactions
  - id (UUID, PK)
  - user_id (UUID, FK → users)
  - feature_type (VARCHAR: 'email_draft', 'proposal', 'summary', 'scoring')
  - input_tokens (INTEGER)
  - output_tokens (INTEGER)
  - created_at (TIMESTAMP)
